class gammapy.astro.source.Pulsar(P_0=<Quantity 0.1 s>, logB=10, n=3, I=<Quantity 1e+45 cm2 g>, R=<Quantity 1000000.0 cm>, age=None, L_0=None, morphology='Delta2D')[source]¶
